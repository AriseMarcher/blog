    自动化引入

    arisemarcher 9/15/2022 vite

    # globEager

    在vite中可以通过import.meta.globEager实现自动导入组件。例如实现组件的自动引入

    文件目录结构

    -- MyProject
src
  components
    Pagination
      index.ts
      index.vue
    otherComponents
      index.ts
      index.vue
main.ts

    相关代码如下

    // main.ts
import { createApp } from 'vue'

const app = createApp(App)

const modules = import.meta.globEager('./components/**/index.ts')

for (const path in modules) {
  app.use(modules[path].default)
}

app.mount('#app')

    // Pagination/index.ts
import { App } from '@vue/runtime-dom'
import Component from './index.vue'

export default {
  install (app: App) {
    app.component('AppPagination', Component)
  }
}
