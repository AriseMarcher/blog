          // 3. åˆ¤æ–­æ˜¯å¦éœ€è¦æ‰§è¡Œå°¾éƒ¨
          if (final && !timer) {
            timer = setTimeout(() => {
              res = fn.apply(this, args)
              resolve(res)
              startTime = new Date().getTime()
              timer = null
            }, waitTime)
          }
        } catch (error) {
          reject(error)
        }
      })
    }

    _throttle.cancel = function() {
      if (timer) clearTimeout(timer)
      startTime = 0
      timer = null
    }

    return _throttle
  }
