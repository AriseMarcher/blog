    防抖

    1. 当事件触发时，【函数】不会立即执行，而是会等待一段时间
    2. 当时间继续频繁触发时，【函数】的执行会被频繁的推迟
    3. 当等待一段时间没有事件触发时，【函数】才会被真正的执行

    精简：【函数】总是在多次点击后、等待了一段时间后才执行。

    应用场景

    • 输入框中频繁输入需要与后台交互
    • 频繁的点击按钮，想要触发某个事件
    • 监听浏览器的滚动事件，完成相关操作
    • 用户缩放浏览器的resize事件

    代码实现

    最基础的实现
    /**
 * @description 防抖函数基础实现
 * @param {Function} fn 需要执行的函数
 * @param {Number} delay 延迟执行时间
 * @returns 
 */
function baseDebounce(fn, delay) {
  // 1. 用于记录上一次事件触发的timer
  let timer = null

  // 2. 触发事件执行的函数
  const _debounce = function () {
    if(timer) clearTimeout(timer)

    timer = setTimeout(() => {
      fn()
      timer = null
    }, delay)
  }

  // 函数返回
  return _debounce
}

    提供立即执行功能
    /**
 * @description 防抖函数基础实现
 * @param {Function} fn 需要执行的函数
 * @param {Number} delay 延迟执行时间
 * @param {Boolean} immediate 是否需要首次立即执行
 * @returns 
 */
function baseDebounce(fn, delay, immediate = false) {
  // 1. 用于记录上一次事件触发的timer
  let timer = null
  let isImmediateExecuted = false

  // 2. 触发事件执行的函数
  const _debounce = function (...args) {
    // 2.1 如果事件多次触发，那么取消上一次的事件
    if(timer) clearTimeout(timer)

    // 2.2 第一次操作不需要延迟，而是立即执行
    if (immediate && !isImmediateExecuted) {
      fn.apply(this, args)
      timer = null
      isImmediateExecuted = true
      return
    }

    timer = setTimeout(() => {
      fn.apply(this, args)
      timer = null
    }, delay)
  }

  // 3. 给_debounce绑定一个取消的函数
  _debounce.cancel = function() {
    if(timer) clearTimeout(timer)
    timer = null
    isImmediateExecuted = false
  }

  return _debounce
}

 * @description 防抖函数基础实现
 * @param {Function} fn 需要执行的函数
 * @param {Number} delay 延迟执行时间
 * @param {Boolean} immediate 是否需要首次立即执行
 * @returns 
 */
function baseDebounce(fn, delay, immediate = false) {
  // 1. 用于记录上一次事件触发的timer
  let timer = null
  let isImmediateExecuted = false

  // 2. 触发事件执行的函数
  const _debounce = function (...args) {
    return new Promise((resolve, reject) => {
      try {
        if(timer) clearTimeout(timer)
      
        // 2.1 如果事件多次触发，那么取消上一次的事件
        let res = null
        if (immediate && !isImmediateExecuted) {
          res = fn.apply(this, args)
          timer = null
          isImmediateExecuted = true
          resolve(res)
          return
        }
        
        // 2.2 第一次操作不需要延迟，而是立即执行
        timer = setTimeout(() => {
          res = fn.apply(this, args)
          timer = null
          isImmediateExecuted = true
          resolve(res)
        }, delay)
      } catch (error) {
        reject(error)
      }
    })
  }

  // 3. 给_debounce绑定一个取消的函数
  _debounce.cancel = function() {
    if(timer) clearTimeout(timer)
    timer = null
    isImmediateExecuted = false
  }

  return _debounce
}

    节流

    1. 当事件触发时，会执行这个事件的响应函数
    2. 如果这个事件会频繁的触发，那么节流函数会按照一定的频率来执行函数
    3. 不管在这个中间有多少次触发这个事件，执行函数的频繁总是固定的。

    精简：【函数】总是在多次点击后、会在一定时间内函数执行的频繁是固定的。

    应用场景：

    • 监听页面的滚动
    • 鼠标移动事件
    • 用户频繁点击按钮操作
    • 游戏中的一些设计

    节流一般都是首次默认执行

    基本实现
    /**
 * 
 * @param {Function} fn 需要执行的函数
 * @param {Number} interVal 有频率执行函数的时间间隔
 * @returns 
 */
function baseThrottle(fn, interVal) {
  let startTime = 0

  const _throttle = function() {
    const nowTime = new Date().getTime()
    const waitTime = interVal - (nowTime - startTime)
    if (waitTime <= 0) {
      fn()
      startTime = nowTime
    }
  }

  return _throttle
}
    控制释放首次执行
    /**
 * 
 * @param {Function} fn 需要执行的函数
 * @param {Number} interVal 有频率执行函数的时间间隔
 * @returns 
 */
function baseThrottle(fn, interVal, immediate = true) {
  let startTime = 0

  const _throttle = function(...args) {
    // 1. 获取当前时间
    const nowTime = new Date().getTime()
    
    // 对立即执行控制，这里主要需要取反，让它直接走下面，默认立即执行
    if (!immediate && startTime === 0) {
      startTime = nowTime
    }

    // 2.计算需要等待的时间执行函数
    const waitTime = interVal - (nowTime - startTime)
    if (waitTime <= 0) {
      fn.apply(this, args)
      startTime = nowTime
    }
  }

  return _throttle
}

    /**
 * 
 * @param {Function} fn 需要执行的函数
 * @param {Number} interVal 有频率执行函数的时间间隔
 * @returns 
 */
function baseThrottle(fn, interVal, {immediate=false, final = false} = {}) {
  let startTime = 0
  let timer = null

  const _throttle = function(...args) {
    // 1. 获取当前时间
    const nowTime = new Date().getTime()
    
    // 对立即执行控制，这里主要需要取反，让它直接走下面，默认立即执行
    if (!immediate && startTime === 0) {
      startTime = nowTime
    }

    // 2.计算需要等待的时间执行函数
    const waitTime = interVal - (nowTime - startTime)
    if (waitTime <= 0) {
      // 极限情况
      if (timer) clearTimeout(timer)
      fn.apply(this, args)
      startTime = nowTime
      timer = null
      return
    }

    // 3. 判断是否需要执行尾部
    if (final && !timer) {
      timer = setTimeout(() => {
        fn.apply(this, args)
        startTime = new Date().getTime()
        timer = null
      }, waitTime)
    }
  }

  return _throttle
}
