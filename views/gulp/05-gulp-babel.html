    # 插件介绍

    • gulp-babel 传送门 (opens new window)
    • @babel/preset-env 传送门 (opens new window)
    • @babel/core 传送门 (opens new window)
    • @babel/plugin-transform-runtime 传送门 (opens new window)
    • @babel/runtime-corejs2 corejs版本2，支持Promise等 传送门 (opens new window)
    • @babel/runtime-corejs3 corejs版本，支持数组includes等 传送门 (opens new window)

    # gulpfile.js 配置

    # babel配置有在gulpfile.js中配置、或在package.json配置，还有创建.babelrc.json文件配置（推荐）

    # 如在gulpfile.js中配置、如下：

    const miniJs = () => {
  return src(`${enterFileName}/**/*.js`)
    .pipe(babel({
      presets: ["@babel/preset-env"],
      plugins: [
        [
          "@babel/plugin-transform-runtime",
          {
            "absoluteRuntime": false,
            "corejs": 3,
            "helpers": true,
            "regenerator": true,
            "version": "7.0.0-beta.0"
          }
        ]
      ]
    }))
    .pipe(dest(`${exportFileName}`))
}

    # .babelrc.json中配置（推荐）

    {
  "presets": ["@babel/preset-env"],
  "plugins": [
    [
      "@babel/plugin-transform-runtime",
      {
        "absoluteRuntime": false,
        "corejs": 3, // 用于指定corejs版本 false：不应用 2：需要安装@babel/runtime-corejs2 3：需要@babel/runtime-corejs3
        "helpers": true,
        "regenerator": true,
        "version": "7.0.0-beta.0"
      }
    ]
  ]
}

    # gulpfile.js的配置为

    const { series, parallel, src, dest } = require('gulp');
// 为了插件更便捷的引入
const gulpLoadPlugins = require('gulp-load-plugins')();

const babel = gulpLoadPlugins.babel;

const enterFileName   = 'code-origin' // 入口文件名
const exportFileName  = 'code-mini'   // 输出文件名

const miniJs = () => {
  return src(`${enterFileName}/**/*.js`)
    .pipe(babel())
    .pipe(dest(`${exportFileName}`))
}

exports.miniJs = miniJs

    # 运行命令

    npm run miniJs

    # 总结
