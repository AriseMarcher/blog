### 安装vue-router

```bash
npm install vue-router@4
```

#### 配置alias
vite.config.ts
```ts
import { fileURLToPath, URL } from 'node:url'

import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
import eslintPlugin from 'vite-plugin-eslint'

// https://vitejs.dev/config/
export default defineConfig({
  plugins: [
    vue(),
    eslintPlugin({
      cache: false
    })
  ],
  resolve: {
    alias: {
      '@': fileURLToPath(new URL('./src', import.meta.url))
    }
  }
})
```

#### router文件配置

router/index.ts

```ts
import {
  createRouter,
  createWebHistory,
  RouteRecordRaw
} from 'vue-router'

// home-index这名称取得也是因为vue-eslint-plugin
// https://eslint.vuejs.org/rules/multi-word-component-names.html
import Home from '@/views/home/home-index.vue'

const routes: Readonly<RouteRecordRaw[]> = [
  {
    path: '/',
    component: Home
  },
  {
    path: '/user',
    // 此处加了 async 和 await是ts语法警告故加上的
    component: async () => await import('@/views/user/user-index.vue')
  }
]

const router = createRouter({
  history: createWebHistory(),
  routes
})

export default router
```

```ts
// router.d.ts
import 'vue-router'

declare module 'vue-router' {
  interface RouteMeta {
    title?: string
    requiresAuth?: boolean
  }
}
```

main.ts配置

```ts
import { createApp } from 'vue'
import './style.css'
import App from './App.vue'
import router from './router/index'

const app = createApp(App)

app.use(router).mount('#app')

```

### 安装vuex

[vuex官网](https://vuex.vuejs.org/zh/)

#### 安装
```bash
npm install vuex@next --save
```

#### vuex.d.ts
```ts
/* eslint-disable @typescript-eslint/no-unused-vars */
import { Store } from 'vuex'
import { ComponentCustomProperties } from 'vue'
import { State } from './store/index'

declare module '@vue/runtime-core' {
  // 为 `this.$store` 提供类型声明
  interface ComponentCustomProperties {
    $store: Store<State>
  }
}
```

#### main.ts

```ts
import { createApp } from 'vue'
import './style.css'
import App from './App.vue'
import router from './router/index'
import { store, key } from './store'

const app = createApp(App)

app.use(store, key)
app.use(router)
app.mount('#app')
```

#### 使用举例

```vue
<template>
  <div>

    </button>

```

### pinia

[pinia官网](https://pinia.web3doc.top/introduction.html)
```bash
npm install pinia
```

#### stores/counter.ts

```ts
import { defineStore } from 'pinia'

// useStore 可以是 useUser、useCart 之类的任何东西
// 第一个参数是应用程序中 store 的唯一 id
export const useCounterStore = defineStore('counterStore', {
  state: () => {
    return {
      counter: 0,
      name: 'alice',
      isAdmin: true
    }
  }
})

```

#### main.ts
```ts
import { createApp } from 'vue'
import './style.css'
import App from './App.vue'
import router from './router/index'
import { store, key } from './store'

import { createPinia } from 'pinia'

const app = createApp(App)
const pinia = createPinia()

app.use(pinia)

app.use(store, key)
app.use(router)
app.mount('#app')
```

#### 使用举例
```vue
<template>
  <div>
    这是User
    <button @click="addCounter">
      点我增加pina点数
    </button>
  </div>
</template>

<script lang="ts" setup>
import { useCounterStore } from '../../stores/counter'

const counterStore = useCounterStore()

const addCounter = () => {
  // counterStore.counter++
  counterStore.$patch({
    counter: counterStore.counter + 1,
    isAdmin: !counterStore.isAdmin
  })
}
</script>
```

+ vuex在ts中推导需要自己写.d.ts的相关配置，pinia无需配置即可支持
+ 二者虽然可以同时使用，但强烈不建议这样做。
